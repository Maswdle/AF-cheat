如何获取聊天记录

使用说明

  1. 安装Python和requests库: pip install requests
  2. 复制下面的代码到Python文件中
  3. 修改代码中的参数:
    • BASE_URL: API的基础URL
    • MY_USER_ID: 你的用户ID
    • USER_TO_ID: 对方用户ID
    • COOKIE_STR: 从浏览器获取的Cookie字符串
  4. 运行脚本获取聊天记录
  5. 将生成的JSON文件上传到本工具查看

Python代码

from urllib.parse import urlparse, parse_qs import requests import json from datetime import datetime import os import re class ChatHistoryFetcher: def __init__(self, base_url, my_user_id, user_to_id, cookies=None): self.base_url = base_url self.my_user_id = my_user_id self.user_to_id = user_to_id self.cookies = cookies or {} self.all_messages = [] self.session = requests.Session() self.other_party_nickname = None self.my_nickname = None def fetch_history(self, first_tid=0, is_first=1): """获取聊天记录""" url = f"{self.base_url}/asmx/method.asmx/randomVIPGetHistoryUserMsgsPage" data = { "myUserID": self.my_user_id, "UserToID": self.user_to_id, "isFirst": is_first, "firstTid": first_tid, "vipcode": "", "serverPort": "1001" } headers = { "accept": "application/json, text/javascript, */*; q=0.01", "accept-language": "zh-CN,zh;q=0.9,en;q=0.8,en-GB;q=0.7,en-US;q=0.6", "content-type": "application/x-www-form-urlencoded; charset=UTF-8", "x-requested-with": "XMLHttpRequest", "referer": f"{self.base_url}/randomdeskrynewmqbqlg.html?v=mqbqlg", "origin": self.base_url } try: response = self.session.post( url, data=data, headers=headers, cookies=self.cookies, timeout=30 ) response.raise_for_status() return response.json() except requests.exceptions.RequestException as e: print(f"请求失败: {e}") return None def get_all_messages(self): """获取所有聊天记录""" first_tid = 0 is_first = 1 page = 1 print("开始获取聊天记录...") while True: print(f"正在获取第 {page} 页记录...") result = self.fetch_history(first_tid, is_first) if not result or "contents_list" not in result or not result["contents_list"]: print("没有更多记录了") break messages = result["contents_list"] self.all_messages.extend(messages) # 更新first_tid为当前页面最后一条记录的Tid first_tid = messages[-1]["Tid"] is_first = 0 # 第一次之后设置为0 page += 1 # 添加延迟避免请求过快 # time.sleep(0.5) # 按时间顺序排序（从早到晚） if self.all_messages: self.all_messages.sort(key=lambda x: datetime.strptime( x["time"], "%Y-%m-%d %H:%M:%S.%f")) # 获取双方昵称 self.my_nickname = self.all_messages[0].get( "nickname") or self.all_messages[0].get("tonickname") self.other_party_nickname = self.all_messages[0].get( "tonickname") or self.all_messages[0].get("nickname") print(f"共获取 {len(self.all_messages)} 条记录") return self.all_messages def generate_filename(self, extension): """生成包含摘要的文件名""" # 获取第一条消息内容作为摘要 first_message_preview = "无消息内容" if self.all_messages: first_content = self.all_messages[0].get("content", "") # 清理内容，移除特殊字符 cleaned_content = re.sub(r'[\\/*?:"<>|]', "", first_content) first_message_preview = cleaned_content[:20] + "..." if len( cleaned_content) > 20 else cleaned_content # 获取当前时间 timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") # 使用双方ID和昵称创建文件名 my_id_short = self.my_user_id[:8] if self.my_user_id else "unknown" other_id_short = self.user_to_id[:8] if self.user_to_id else "unknown" my_nickname_short = re.sub( r'[\\/*?:"<>|]', "", self.my_nickname or "我")[:10] other_nickname_short = re.sub( r'[\\/*?:"<>|]', "", self.other_party_nickname or "对方")[:10] filename = f"chat_{my_id_short}_{other_id_short}_{my_nickname_short}_{other_nickname_short}_{first_message_preview}_{timestamp}.{extension}" return filename def export_to_json(self, filename=None): """导出为JSON文件""" if not filename: filename = self.generate_filename("json") # 添加元数据 export_data = { "metadata": { "export_time": datetime.now().isoformat(), "my_user_id": self.my_user_id, "user_to_id": self.user_to_id, "my_nickname": self.my_nickname, "other_party_nickname": self.other_party_nickname, "total_messages": len(self.all_messages), "time_range": { "start": self.all_messages[0]["time"] if self.all_messages else None, "end": self.all_messages[-1]["time"] if self.all_messages else None } }, "messages": self.all_messages } with open(filename, 'w', encoding='utf-8') as f: json.dump(export_data, f, ensure_ascii=False, indent=2) print(f"已导出JSON文件: {filename}") return filename def export_to_text(self, filename=None): """导出为文本文件""" if not filename: filename = self.generate_filename("txt") with open(filename, 'w', encoding='utf-8') as f: # 写入元数据 f.write("聊天记录导出\n") f.write("=" * 50 + "\n") f.write(f"导出时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n") f.write(f"我的ID: {self.my_user_id}\n") f.write(f"我的昵称: {self.my_nickname or '未知'}\n") f.write(f"对方ID: {self.user_to_id}\n") f.write(f"对方昵称: {self.other_party_nickname or '未知'}\n") f.write(f"消息总数: {len(self.all_messages)}\n") if self.all_messages: f.write( f"时间范围: {self.all_messages[0]['time']} 至 {self.all_messages[-1]['time']}\n") f.write("=" * 50 + "\n\n") # 写入消息内容 for msg in self.all_messages: time_str = msg["time"] sender = msg["nickname"] receiver = msg["tonickname"] content = msg["content"] f.write(f"[{time_str}] {sender} -> {receiver}: {content}\n") print(f"已导出文本文件: {filename}") return filename def extract_user_id(url): """ 从URL中提取touserid参数的值 """ parsed_url = urlparse(url) query_params = parse_qs(parsed_url.query) return query_params.get('touserid', [None])[0] def load_my_user_id(): """从文件加载我的用户ID""" if os.path.exists("my_user_id.txt"): try: with open("my_user_id.txt", "r", encoding="utf-8") as f: return f.read().strip() except: pass return None def save_my_user_id(user_id): """保存我的用户ID到文件""" with open("my_user_id.txt", "w", encoding="utf-8") as f: f.write(user_id) def get_cookies(): """获取cookies""" # 从浏览器获取的Cookie字符串 COOKIE_STR = """ room_name_random=%E5%85%AC%E5%85%B1%E5%9C%BA%E6%89%80; room_psw_random=; userAge=0; switchvipsex=0; switchVipAddress=0; switchHealthMode=0; isMoon=1; colorpicker=%23414550; isSound=-1; randomVipCode=; user_id=fe077f0b39cb4c0ebac68a35e7fc16b6; user_nickname_random=%E6%83%B3%F0%9F%90%A2%E9%AB%98%E4%B8%AD%E5%A5%B3%E5%8F%8B%E5%AE%B6%E7%AE%A1%E4%B8%A5 """ # 将Cookie字符串转换为字典 cookies = {} for cookie in COOKIE_STR.strip().split(';'): if '=' in cookie: key, value = cookie.strip().split('=', 1) cookies[key] = value return cookies # 使用示例 if __name__ == "__main__": # 配置参数 BASE_URL = "http://45.119.55.75:668" # 尝试加载已保存的用户ID MY_USER_ID = load_my_user_id() # 如果没有保存的用户ID，则要求用户输入 if not MY_USER_ID: MY_USER_ID = input("请输入您的用户ID: ").strip() save_my_user_id(MY_USER_ID) print(f"已保存您的用户ID: {MY_USER_ID}") else: print(f"使用已保存的用户ID: {MY_USER_ID}") change = input("是否要更改用户ID? (y/N): ").strip().lower() if change == 'y': MY_USER_ID = input("请输入新的用户ID: ").strip() save_my_user_id(MY_USER_ID) print(f"已更新您的用户ID: {MY_USER_ID}") # 获取cookies cookies = get_cookies() # 循环处理多个聊天链接 while True: print("\n" + "="*50) url = input("请输入聊天链接(直接回车退出): ").strip() if not url: print("退出程序") break USER_TO_ID = extract_user_id(url) if not USER_TO_ID: print("无法从链接中提取用户ID，请检查链接格式") continue print(f"提取到对方用户ID: {USER_TO_ID}") # 创建抓取器实例 fetcher = ChatHistoryFetcher(BASE_URL, MY_USER_ID, USER_TO_ID, cookies) # 获取所有聊天记录 messages = fetcher.get_all_messages() if messages: # 导出文件 json_file = fetcher.export_to_json() text_file = fetcher.export_to_text() print(f"\n导出完成!") print(f"JSON文件: {os.path.abspath(json_file)}") print(f"文本文件: {os.path.abspath(text_file)}") else: print("未能获取到聊天记录，请检查参数和Cookie") # 询问是否继续 continue_choice = input("是否继续处理其他链接?
