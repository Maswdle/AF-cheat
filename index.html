如何获取聊天记录

使用说明

  1. 安装Python和requests库: pip install requests
  2. 复制下面的代码到Python文件中
  3. 修改代码中的参数:
    • BASE_URL: API的基础URL
    • MY_USER_ID: 你的用户ID
    • USER_TO_ID: 对方用户ID,要注意的是,你不要知道对方的实际身份码是多少,只需要知道你能看到的ID(实际上是实际身份码经过MD5密后的结果)
    • COOKIE_STR: 从浏览器获取的Cookie字符串
  4. 运行脚本获取聊天记录
  5. 将生成的JSON文件上传到本工具查看

Python代码

import requests import json from datetime import datetime import os class ChatHistoryFetcher: def __init__(self, base_url, my_user_id, user_to_id, cookies=None): self.base_url = base_url self.my_user_id = my_user_id self.user_to_id = user_to_id self.cookies = cookies or {} self.all_messages = [] self.session = requests.Session() def fetch_history(self, first_tid=0, is_first=1): """获取聊天记录""" url = f"{self.base_url}/asmx/method.asmx/randomVIPGetHistoryUserMsgsPage" data = { "myUserID": self.my_user_id, "UserToID": self.user_to_id, "isFirst": is_first, "firstTid": first_tid, "vipcode": "", "serverPort": "1001" } headers = { "accept": "application/json, text/javascript, */*; q=0.01", "accept-language": "zh-CN,zh;q=0.9,en;q=0.8,en-GB;q=0.7,en-US;q=0.6", "content-type": "application/x-www-form-urlencoded; charset=UTF-8", "x-requested-with": "XMLHttpRequest", "referer": f"{self.base_url}/randomdeskrynewmqbqlg.html?v=mqbqlg", "origin": self.base_url } try: response = self.session.post( url, data=data, headers=headers, cookies=self.cookies, timeout=30 ) response.raise_for_status() return response.json() except requests.exceptions.RequestException as e: print(f"请求失败: {e}") return None def get_all_messages(self): """获取所有聊天记录""" first_tid = 0 is_first = 1 page = 1 print("开始获取聊天记录...") while True: print(f"正在获取第 {page} 页记录...") result = self.fetch_history(first_tid, is_first) if not result or "contents_list" not in result or not result["contents_list"]: print("没有更多记录了") break messages = result["contents_list"] self.all_messages.extend(messages) first_tid = messages[-1]["Tid"] is_first = 0 page += 1 # 延迟 # time.sleep(0.5) self.all_messages.sort(key=lambda x: datetime.strptime( x["time"], "%Y-%m-%d %H:%M:%S.%f")) print(f"共获取 {len(self.all_messages)} 条记录") return self.all_messages def export_to_json(self, filename=None): """导出为JSON文件""" if not filename: timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") filename = f"chat_history_{timestamp}.json" with open(filename, 'w', encoding='utf-8') as f: json.dump(self.all_messages, f, ensure_ascii=False, indent=2) print(f"已导出JSON文件: {filename}") return filename def export_to_text(self, filename=None): """导出为文本文件""" if not filename: timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") filename = f"chat_history_{timestamp}.txt" with open(filename, 'w', encoding='utf-8') as f: for msg in self.all_messages: # 格式化每条消息 time_str = msg["time"] sender = msg["nickname"] receiver = msg["tonickname"] content = msg["content"] f.write(f"[{time_str}] {sender} -> {receiver}: {content}\n") print(f"已导出文本文件: {filename}") return filename if __name__ == "__main__": # 配置参数 BASE_URL = "http://45.119.55.75:668" MY_USER_ID = "277b7dbfa44f4314af65ce99154761a0" # 替换为你的用户ID USER_TO_ID = "deff5f25adea714203f520dc0169dfd6" # 替换为对方用户ID COOKIE_STR = """ room_name_random=%E5%85%AC%E5%85%B1%E5%9C%BA%E6%89%80; room_psw_random=; userAge=0; switchvipsex=0; switchVipAddress=0; switchHealthMode=0; isMoon=1; colorpicker=%23414550; isSound=-1; randomVipCode=; user_id=YOURID; user_nickname_random=ss """ cookies = {} for cookie in COOKIE_STR.strip().split(';'): if '=' in cookie: key, value = cookie.strip().split('=', 1) cookies[key] = value fetcher = ChatHistoryFetcher(BASE_URL, MY_USER_ID, USER_TO_ID, cookies) messages = fetcher.get_all_messages() if messages: json_file = fetcher.export_to_json() text_file = fetcher.export_to_text() print(f"\n导出完成!") print(f"JSON文件: {os.path.abspath(json_file)}") print(f"文本文件: {os.path.abspath(text_file)}") else: print("未能获取到聊天记录，请检查参数和Cookie")
